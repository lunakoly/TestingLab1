183 fun fromRoman(roman: String): Int {
184     val rToN = mapOf('I' to 1, 'V' to 5, 'X' to 10, 'L' to 50, 'C' to 100, 'D' to 500, 'M' to 1000)
185     var number: Int
186     try {
187         if (roman.isNotEmpty()) {
188             try {
189                 number = rToN[roman[0]]!!
190             } catch (e: NullPointerException) {
191                 throw IllegalArgumentException()
192             }
193         } else throw IllegalArgumentException()
194         var sameNum = 0
195         for (i in 1 until roman.length) {
196             val v1 = rToN[roman[i - 1]]
197             val v2 = rToN[roman[i]]
198             if (v1 == null || v2 == null) throw IllegalArgumentException()
199             if (v1 > v2) {
200                 number += v2
201                 sameNum = 0
202             }
203             if (v1 == v2) {
204                 number += v2
205                 sameNum += 1
206                 if (sameNum > 3) throw IllegalArgumentException()
207             }
208             if (v1 < v2 && v2 > number) {
209                 number = v2 - number
210                 if (sameNum > 1) throw IllegalArgumentException()
211                 sameNum = 0
212             }
213             if (v1 < v2 && v2 < number) {
214                 number = number + v2 - 2 * v1
215                 if (sameNum > 1) throw IllegalArgumentException()
216                 sameNum = 0
217             }
218         }
219     } catch (e: IllegalArgumentException) {
220         return -1
221     }
222     return (number)

261 fun computeDeviceCells(cells: Int, commands: String, limit: Int): List<Int> {
262     val device = MutableList(cells) { 0 }
263     var comCount = 0
264     val cycleBegins = mutableListOf(0)
265     var n = 0
266     var cellsN = cells / 2
267     var comN = 0
268     var skip = false
269     var skipThrough = 0
270     var bCount = 0
271     var eCount = 0
272     for (i in commands.indices) {
273         if (commands[i] != '>' && commands[i] != '<' && commands[i] != '+' && commands[i] != '-'
274             && commands[i] != '[' && commands[i] != ']' && commands[i] != ' '
275         ) throw IllegalArgumentException()
276         if (commands[i] == '[') bCount += 1
277         if (commands[i] == ']') eCount += 1
278         if (eCount > bCount) throw IllegalArgumentException()
279     }
280     if (bCount != eCount) throw IllegalArgumentException()
281     while (comN < commands.length) {
282         val com = commands[comN]
283         if (skip && com == '[') skipThrough += 1
284         if (!skip) {
285             if (com == '>') cellsN += 1
286             if (com == '<') cellsN -= 1
287             if (cellsN > cells - 1 || cellsN < 0) throw IllegalStateException()
288             if (com == '+') device[cellsN] += 1
289             if (com == '-') device[cellsN] -= 1
290             if (com == '[') {
291                 if (device[cellsN] == 0) skip = true
292                 else {
293                     if (cycleBegins[n] == 0) cycleBegins[n] = comN
294                     else {
295                         n += 1
296                         cycleBegins.add(n, comN)
297                     }
298                 }
299             }
300             if (com == ']') {
301                 if (device[cellsN] != 0) comN = cycleBegins[n]
302                 else {
303                     cycleBegins[n] = 0
304                     if (n != 0) n -= 1
305                 }
306             }
307         }
308         if (skip && com == ']') {
309             if (skipThrough == 0) skip = false
310             else skipThrough -= 1
311         }
312         if (!skip) comCount += 1
313         if (comCount == limit) break
