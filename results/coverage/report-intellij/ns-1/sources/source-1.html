275  *

277  * используя то, что вы узнали в данном уроке.
278  *
279  * Например:
280  *   findSumOfTwo(listOf(1, 2, 3), 4) -> Pair(0, 2)
281  *   findSumOfTwo(listOf(1, 2, 3), 6) -> Pair(-1, -1)
282  */
283 fun findSumOfTwo(list: List<Int>, number: Int): Pair<Int, Int> {
284     val nmap = mutableMapOf<Int, Int>()
285     for (i in list.indices) {
286         if (!nmap.containsKey(list[i])) nmap[list[i]] = i
287         else if (list[i] * 2 == number) return Pair(nmap[list[i]]!!, i)
288     }
289     for ((n, ind) in nmap)
290         if (nmap.contains(number - n) && n * 2 != number) return Pair(ind, nmap[number - n]!!)
291 
292     return Pair(-1, -1)
293 }

318 fun bagPacking(treasures: Map<String, Pair<Int, Int>>, capacity: Int): Set<String> {
319     val results = mutableMapOf<Pair<Int, Int>, Int>()
320     val weight = MutableList(treasures.size) { 0 }
321     val value = MutableList(treasures.size) { 0 }
322     val t = MutableList(treasures.size) { "" }
323     var lowWeight = capacity
324     val neededT = mutableMapOf<Pair<Int, Int>, Set<String>>()
325     var k = 1
326     for ((name, vw) in treasures) {
327         weight.add(k, vw.first)
328         value.add(k, vw.second)
329         t.add(k, name)
330         if (lowWeight > vw.first) lowWeight = vw.first
331         k += 1
332     }
333     for (j in 1..capacity) results[Pair(0, j)] = 0
334     for (i in 1..treasures.size) {
335         if (lowWeight <= capacity && treasures.isNotEmpty()) {
336             for (j in lowWeight..capacity) {
337                 if (weight[i] > j) {
338                     results[Pair(i, j)] = results[Pair(i - 1, j)]!!
339                     neededT[Pair(i, j)] = neededT[Pair(i - 1, j)] ?: setOf()
340                 } else {
341                     if (results[Pair(i - 1, j)] == 0) {
342                         results[Pair(i, j)] = value[i]
343                         neededT[Pair(i, j)] = setOf(t[i])
344                     } else {
345                         if (results[Pair(i - 1, j - weight[i])] == null) {
346                             results[Pair(i, j)] = max(results[Pair(i - 1, j)]!!, value[i])
347                             if (value[i] > results[Pair(i - 1, j)]!!) neededT[Pair(i, j)] = setOf(t[i])
348                             if (value[i] <= results[Pair(i - 1, j)]!!) neededT[Pair(i, j)] = neededT[Pair(i - 1, j)]!!
349                         } else {
350                             results[Pair(i, j)] = max(
351                                 results[Pair(i - 1, j)]!!, results[Pair(i - 1, j - weight[i])]!! + value[i]
352                             )
353                             if (results[Pair(i - 1, j - weight[i])]!! + value[i] > results[Pair(i - 1, j)]!!) {
354                                 neededT[Pair(i, j)] = neededT[Pair(i - 1, j - weight[i])]!! + t[i]
355                             } else neededT[Pair(i, j)] = neededT[Pair(i - 1, j)]!!
356                         }
357                     }
358                 }
359             }
360         }
361     }
362     return if (neededT[Pair(treasures.size, capacity)] != null) neededT[Pair(treasures.size, capacity)]!!
363     else emptySet()
364 }
