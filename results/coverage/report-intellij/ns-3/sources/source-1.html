24 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
25     val writer = File(outputName).bufferedWriter()
26     var currentLineLength = 0
27     fun append(word: String) {
28         if (currentLineLength > 0) {
29             if (word.length + currentLineLength >= lineLength) {
30                 writer.newLine()
31                 currentLineLength = 0
32             } else {
33                 writer.write(" ")
34                 currentLineLength++
35             }
36         }
37         writer.write(word)
38         currentLineLength += word.length
39     }
40     for (line in File(inputName).readLines()) {
41         if (line.isEmpty()) {
42             writer.newLine()
43             if (currentLineLength > 0) {
44                 writer.newLine()
45                 currentLineLength = 0
46             }
47             continue
48         }
49         for (word in line.split(Regex("\\s+"))) {
50             append(word)
51         }
52     }
53     writer.close()
54 }

88     val results = mutableMapOf<String, Int>()
89     val text = File(inputName).readText()
90     var begInd: Int
91     for (n in substrings.indices) {
92         val subS = substrings[n]
93         var reg = Regex(subS, setOf(RegexOption.IGNORE_CASE, RegexOption.LITERAL)).find(text)
94         if (reg == null) results[subS] = 0
95         else {
96             results[subS] = 1
97             begInd = reg.range.first + 1
98             while (reg != null) {
99                 reg = Regex(subS, setOf(RegexOption.IGNORE_CASE, RegexOption.LITERAL)).find(text, begInd)
100                 if (reg == null) break
101                 results[subS] = results[subS]!! + 1
102                 begInd = reg.range.first + 1
103             }
104         }
105     }
106     return results
107 }

185 fun alignFileByWidth(inputName: String, outputName: String) {
186     val lines = mutableListOf<Pair<String, Int>>()
187     for (line in File(inputName).readLines()) {
188         var newL = line.trim()
189         newL = newL.replace(Regex("\\s+"), " ")
190         val wordCount = Regex("\\S+\\s").findAll(newL).count() + 1
191         lines.add(Pair(newL, wordCount))
192     }
193     var maxL = 0
194     for ((first) in lines) if (first.length > maxL) maxL = first.length
195     val writer = File(outputName).bufferedWriter()
196     writer.use {
197         for ((line, wCount) in lines) {
198             var needToAdd: Int
199             if (maxL != line.length && wCount > 1) needToAdd = (maxL - line.length).mod(wCount - 1)
200             else {
201                 writer.write(line)
202                 writer.newLine()
203                 continue
204             }
205             val add = (maxL - line.length) / (wCount - 1)
206             if (needToAdd == 0) {
207                 for (i in line.indices) {
208                     if (line[i] == ' ') for (k in 1..add) writer.write(" ")
209                     writer.write(line[i].toString())
210                 }
211             } else {
212                 for (i in line.indices) {
213                     if (line[i] == ' ') {
214                         for (k in 1..add) writer.write(" ")
215                         if (needToAdd != 0) {
216                             writer.write(" ")
217                             needToAdd -= 1
218                         }
219                     }
220                     writer.write(line[i].toString())
221                 }
222             }
223             writer.newLine()
224         }
225     }
226 }
